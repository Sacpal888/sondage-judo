async function loadAndApplyDesign() { try { const { data, error } = await supabaseClient .from('settings') .select('*') .in('key', ['design_title_color', 'design_text_color', 'design_font_family', 'design_bg_color', 'design_border_color']); if (error) return; const design = {}; if (data) { data.forEach(item => { design[item.key] = item.value; }); } if (design.design_title_color || design.design_text_color || design.design_font_family || design.design_bg_color || design.design_border_color) { const style = document.createElement('style'); let css = ''; if (design.design_title_color) { css += 'h1, h2, h3, .player-name, .survey-title { color: ' + design.design_title_color + ' !important; }\n'; } if (design.design_text_color) { css += 'body, p, span, div, .stat-label { color: ' + design.design_text_color + ' !important; }\n'; } if (design.design_font_family) { css += 'body, * { font-family: ' + design.design_font_family + ' !important; }\n'; } if (design.design_bg_color) { css += '.player-card, .survey-card, .result-card { background: ' + design.design_bg_color + ' !important; }\n'; } if (design.design_border_color) { css += '.player-card, .survey-card, .result-card { border-color: ' + design.design_border_color + ' !important; }\n'; } style.textContent = css; document.head.appendChild(style); } } catch (error) { console.error('Erreur chargement design:', error); } } window.addEventListener('DOMContentLoaded', loadAndApplyDesign);
